需要通过中间寄存器(ax)赋值的寄存器：段寄存器（CS, DS, SS, ES(Extra)）

**堆栈传参**

```assembly
f:
   push bp
   mov bp, sp
   mov ax, [bp+4]; 从堆栈中取得参数
   add ax, ax
   pop bp
   ret
main:
   mov ax, 3
   push ax; 参数压入到堆栈
   call f
   add sp, 2
```

压入bp保护，然后把栈顶偏移地址传给bp，然后+4可以跳过栈顶的bp和次顶的ip拿到参数(且由于是用mov=因此不会改变栈顶的位置)，出来到main之后再把sp+2可以使栈顶的3被略过

==[bp]默认段地址是ss，剩下三个都是ds==

**栈底位置很大(压入第一个之后sp从0000h变为0FFFEh)，进栈sp减小，出栈sp增大**

**动态变量**

```c
int f(int a, int b) {
	int c; /* c是局部动态变量 */
	c = a+b;
	return c;
}
```

上述C语言函数可翻译成以下汇编代码:

```assembly
f:
	push bp; (4)
	mov bp, sp
	sub sp, 2; (5) 这里挖的坑就是给变量c的
	mov ax, [bp+4]
	add ax, [bp+6]
	mov [bp-2], ax
	;mov ax, [bp-2] 多写了？？
	mov sp, bp; (6)此时变量c死亡 ；相当于 `add sp,2`
	pop bp; (7)
ret; (8)

main:
	mov ax, 20
	push ax; (1)
	mov ax, 10
	push ax; (2)
	call f; (3)
	here:
	add sp, 4;(9)此时参数a,b死亡 ; but c还在ss:[sp-A]处？ 是的
```

执行上述代码时, 堆栈布局如下:

```assembly
ss:1FF6 [30]     (5) 变量c
ss:1FF8 old bp   <- bp(4)(6)
ss:1FFA here IP  <- (3)(7)
ss:1FFC 10       <- (2)(8)
ss:1FFE 20       <- (1)
ss:2000          <-(9)
```

在函数`f()`中时，bp指向bp的地址，bp+2指向here的IP，+2和+4分别指向b，a，-2指向c

**递归**

```c
int f(int n) {
	if(n==1)
		return 1;
	return n+f(n-1);
}
```

上述C语言递归函数可翻译成以下汇编代码:

```assembly
f:
	push bp;(3)(6)(9)
	mov bp, sp
	mov ax, [bp+4] ;*bp = bp; *(bp+2) = IP;
	cmp ax, 1
	je done
	dec ax
	push ax	;(4)(7) 传n-1作为参数进入下一层递归
	call f	;(5)(8)
	there:
	add sp, 2;(12)(15); 刷掉**传进去**的参数
	add ax, [bp+4] ; 这是经历过递归的bp不是上面的了，= [sp+2] = f(n-1)
	done:
	pop bp;(10)(13)(16); 丢弃本次函数中用的bp
ret;(11)(14)(17)

main:
	mov ax, 3
	push ax;(1)
	call f;(2)
	here:
	add sp, 2;(18)
```

执行上述代码时的堆栈布局如下:

```assembly
ss:1FEE oldbp<-bp(9)
ss:1FF0 there<-(8)(10)
ss:1FF2 1<-(7)(11)
ss:1FF4 oldbp<-bp(6)(12)
ss:1FF6 there<-(5)(13)
ss:1FF8 2<-(4)(14)
ss:1FFA oldbp<-bp(3)(15)
ss:1FFC here <-(2)(16)
ss:1FFE 3 <-(1)(17)
ss:2000   <-(18)
```
